<script>

function calc(){

let harga = parseFloat(document.getElementById('loan').value);

let dp    = parseFloat(document.getElementById('dp').value);

let years = parseInt(document.getElementById('years').value);

let rate  = parseFloat(document.getElementById('rate').value)/100;

if(isNaN(harga) || isNaN(dp) || harga<=dp){

document.getElementById('output').innerText = 'Isi harga & DP dengan benar';

return;

}

let principal = harga - dp;

let i = rate/12;

let n = years*12;

let monthly = principal * (i * Math.pow(1+i,n)) / (Math.pow(1+i,n)-1);

document.getElementById('output').innerText =

Cicilan per bulan: Rp  + monthly.toLocaleString('id-ID',{maximumFractionDigits:0});

}

</script>
